Lua Client Example

Fetches all builds from the API, then downloads each one.

-- Luau / Roblox version
local HttpService = game:GetService("HttpService")

local API_URL = "https://sarhossamet1.github.io/storage/builds.json"

-- Step 1: get all available builds
local response = HttpService:GetAsync(API_URL)
local data = HttpService:JSONDecode(response)

print("Found " .. data.total .. " builds:")

-- Step 2: download each build
for _, build in ipairs(data.builds) do
    print("Downloading: " .. build.name)
    local json = HttpService:GetAsync(build.url)
    -- save / process the build data here
    print("  Done (" .. build.size .. " bytes)")
end
